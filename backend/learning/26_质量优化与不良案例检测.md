```python
    # 获取消息和会话详情
```

### 26.3 重复提问检测

```python
def detect_from_repeated_query(
    self, session_id, user_id, enterprise_id, query, previous_responses
):
    """
    检测重复提问
    
    算法逻辑：
    1. 记录用户近期提问
    2. 检查时间窗口内（默认5分钟）
    3. 如果同一（或相似）问题出现次数
    4. 超过阈值（默认3次）触发
    """
    query_key = f"{user_id}:{query[:50].lower()}"
    
    # 记录当前查询
    self.recent_queries[query_key].append({
        "session_id": str(session_id),
        "timestamp": datetime.utcnow(),
        "query": query
    })
    
    # 清理过期记录
    recent = [
        q for q in self.recent_queries[query_key]
        if (datetime.utcnow() - q["timestamp"]).total_seconds() < 300
    ]
    
    if len(recent) >= 3:
        return BadCase(
            case_type=BadCaseType.REPEATED_QUERY,
            severity=BadCaseSeverity.MEDIUM,
            reasons=[f"用户重复提问 {len(recent)} 次"],
            ...
        )
```

### 26.4 低检索得分检测

```python
def detect_from_low_score(
    self, session_id, user_id, enterprise_id, query, ai_response, retrieval_score
):
    """
    从检索得分判断
    
    严重程度映射：
    - score < 0.1 → HIGH
    - score < 0.2 → MEDIUM
    - score < 0.3 → LOW
    """
    if retrieval_score >= 0.3:
        return None
    
    # 根据分数确定严重程度
    if retrieval_score < 0.1:
        severity = BadCaseSeverity.HIGH
    elif retrieval_score < 0.2:
        severity = BadCaseSeverity.MEDIUM
    else:
        severity = BadCaseSeverity.LOW
    
    return BadCase(
        case_type=BadCaseType.LOW_RETRIEVAL_SCORE,
        severity=severity,
        reasons=[f"检索得分过低: {retrieval_score:.2f}"],
        metadata={"retrieval_score": retrieval_score},
        ...
    )
```

## 4. 存储与管理

### 26.1 BadCaseRepository 仓库类

```python
class BadCaseRepository:
    """Bad case 存储管理
    
    存储策略：
    - Redis 存储，支持快速读写
    - 按企业ID索引
    - 30天保留期
    - 支持状态查询和统计
    """
    
    async def save_bad_case(self, bad_case):
        """保存 bad case 到 Redis"""
        key = f"badcase:{bad_case.case_id}"
        await self.redis.set(
            key, json.dumps(bad_case.to_dict()), ex=86400*30
        )
        
        # 添加到企业的列表
        list_key = f"badcase:list:{bad_case.enterprise_id}"
        await self.redis.lpush(list_key, bad_case.case_id)
```

### 26.2 查询与统计

```python
async def get_bad_cases_by_enterprise(
    self, enterprise_id, status=None, limit=50
):
    """
    获取企业的 bad cases
    
    支持按状态过滤：
    - pending_review: 待审核
    - reviewed: 已审核
    """
    list_key = f"badcase:list:{enterprise_id}"
    case_ids = await self.redis.lrange(list_key, 0, limit-1)
    
    cases = []
    for case_id in case_ids:
        key = f"badcase:{case_id}"
        data = await self.redis.get(key)
        if data:
            case = json.loads(data)
            if status is None or case.get("status") == status:
                cases.append(case)
    return cases

async def get_bad_case_stats(self, enterprise_id):
    """获取统计数据"""
    cases = await self.get_bad_cases_by_enterprise(...)
    
    return {
        "total": len(cases),
        "pending_review": len([c for c in cases if ...],
        "by_type": {...},
        "by_severity": {...}
    }
```

### 26.1 多维度信号融合

```python
# 综合多个信号综合判断
def calculate_overall_severity(bad_cases):
    """
    综合多个信号综合判断
    """
    score = 0
    
    for case in bad_cases:
        if case.case_type == BadCaseType.EXPLICIT_THUMBS_DOWN:
            score += 3
        elif case.case_type == BadCaseType.HUMAN_TAKEOVER:
            score += 5
        elif case.case_type == BadCaseType.REPEATED_QUERY:
                score += 2
        elif case.case_type == BadCaseType.LOW_RETRIEVAL_SCORE:
                score += 1
    
    if score >= 5:
        return BadCaseSeverity.CRITICAL
    elif score >= 3:
        return BadCaseSeverity.HIGH
    elif score >= 1:
        return BadCaseSeverity.MEDIUM
    else:
        return BadCaseSeverity.LOW
```

```python
def analyze_bad_cases(enterprise_id):

    # 严重程度分布

        severity = case["severity"]
        severity_counts[severity] = severity_counts.get(severity, 0) + 1

    return {
        "type_distribution": type_counts,
        "severity_distribution": severity_counts,
        "top_queries": sorted(query_counts.items(), key=lambda x: -x[1])[:10]
```
