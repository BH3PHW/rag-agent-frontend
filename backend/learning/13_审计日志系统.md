```

法律合规：

✅ 《个人信息保护法》- 数据处理需记录审计
✅ 《等保2.0》- 三级系统必须具备审计功能
```

| 特性 | 普通日志 | 审计日志 |

| **内容** | 技术信息 | 业务操作详情 |

| **保留** | 短期 | 长期（至少6个月） |

## 13.3 审计日志记录

### 13.3.1 基础记录器

**工程化实现**：高性能缓冲写入

```python
class AuditLogger:
    """审计日志记录器"""

    def __init__(self, max_buffer_size: int = 100):
        """
        参数:
            max_buffer_size: 缓冲区大小，累积到一定数量后写入磁盘
        """
        self.max_buffer_size = max_buffer_size
        self.buffer: List[AuditLog] = []  # 内存缓冲区
        self.lock = threading.Lock()       # 线程安全
        self.enabled = True

    def log(
        self,
        action: AuditAction,
        user_id: Optional[str] = None,
        enterprise_id: Optional[str] = None,
        severity: AuditSeverity = AuditSeverity.INFO,
        success: bool = True,
        **kwargs
    ) -> str:
        """记录审计日志"""
        if not self.enabled:
            return None

        # 生成唯一ID
        log_id = str(uuid.uuid4())

        # 创建审计日志对象
        audit_log = AuditLog(
            id=log_id,
            timestamp=datetime.utcnow(),
            action=action.value,
            user_id=user_id,
            enterprise_id=enterprise_id,
            severity=severity.value,
            success=success,
            **kwargs
        )

        # 加入缓冲区
        with self.lock:
            self.buffer.append(audit_log)

            # 缓冲区满了，刷新到磁盘
            if len(self.buffer) >= self.max_buffer_size:
                self._flush()

        return log_id
```

### 13.3.2 批量刷新机制

```python
def _flush(self):
    """刷新缓冲区到存储"""
    if not self.buffer:
        return

    logs_to_write = self.buffer.copy()
    self.buffer.clear()

    # 实际实现：写入文件/数据库/Elasticsearch
    for log in logs_to_write:
        # 1. 写入数据库（推荐：便于查询分析）
        db.insert("audit_logs", log.to_dict())

        # 2. 写入文件（备份）
        write_to_file(log.to_json())

        # 3. 发送到Elasticsearch（可选：高级搜索）
        send_to_elasticsearch(log)
```

```python
class SecurityAuditor:
    """安全审计员 - 封装常见安全审计操作"""

    def __init__(self, audit_logger: AuditLogger):
        self.logger = audit_logger

    def audit_login(
        self,
        user_id: str,
        enterprise_id: str,
        success: bool,
        ip_address: Optional[str] = None,
        error_message: Optional[str] = None
    ):
        """登录审计"""
        action = AuditAction.LOGIN if success else AuditAction.LOGIN_FAILED
        severity = AuditSeverity.INFO if success else AuditSeverity.WARNING

        self.logger.log(
            action=action,
            user_id=user_id,
            enterprise_id=enterprise_id,
            severity=severity,
            success=success,
            ip_address=ip_address,
            error_message=error_message
        )

    def audit_data_access(
        self,
        action: AuditAction,
        user_id: str,
        enterprise_id: str,
        resource_type: str,
        resource_id: str,
        success: bool,
        platform: Optional[str] = None
    ):
        """数据访问审计"""
        severity = AuditSeverity.INFO if success else AuditSeverity.WARNING

        self.logger.log(
            action=action,
            user_id=user_id,
            enterprise_id=enterprise_id,
            severity=severity,
            success=success,
            resource_type=resource_type,
            resource_id=resource_id,
            platform=platform
        )

    def audit_security_event(
        self,
        event_type: str,
        user_id: Optional[str],
        enterprise_id: Optional[str],
        details: Dict[str, Any],
        severity: AuditSeverity = AuditSeverity.WARNING
    ):
        """安全事件审计"""
        self.logger.log(
            action=AuditAction.SECURITY_BLOCK,
            user_id=user_id,
            enterprise_id=enterprise_id,
            severity=severity,
            success=False,
            metadata={"event_type": event_type, **details}
        )

    def audit_permission_denied(
        self,
        user_id: str,
        enterprise_id: str,
        resource_type: str,
        resource_id: str,
        reason: str
    ):
        """权限拒绝审计"""
        self.logger.log(
            action=AuditAction.PERMISSION_DENIED,
            user_id=user_id,
            enterprise_id=enterprise_id,
            severity=AuditSeverity.WARNING,
            success=False,
            resource_type=resource_type,
            resource_id=resource_id,
            error_message=reason
        )
```

## 13.5 查询与分析

### 13.5.1 日志查询

**产品化思维**：支持多种查询场景

```python
def query(
    self,
    user_id: Optional[str] = None,
    enterprise_id: Optional[str] = None,
    action: Optional[str] = None,
    start_time: Optional[datetime] = None,
    end_time: Optional[datetime] = None,
    severity: Optional[str] = None,
    limit: int = 100
) -> List[AuditLog]:
    """查询审计日志"""
    # 构建查询条件
    query = "SELECT * FROM audit_logs WHERE 1=1"
    params = []

    if user_id:
        query += " AND user_id = ?"
        params.append(user_id)

    if enterprise_id:
        query += " AND enterprise_id = ?"
        params.append(enterprise_id)

    if action:
        query += " AND action = ?"
        params.append(action)

    if start_time:
        query += " AND timestamp >= ?"
        params.append(start_time)

    if end_time:
        query += " AND timestamp <= ?"
        params.append(end_time)

    if severity:
        query += " AND severity = ?"
        params.append(severity)

    query += f" ORDER BY timestamp DESC LIMIT {limit}"

    return db.execute(query, params)
```

### 13.5.2 用户活动统计

```python
def get_user_activity(
    self,
    user_id: str,
    enterprise_id: Optional[str] = None,
    days: int = 7
) -> Dict[str, Any]:
    """获取用户活动统计"""
    # 查询最近N天的操作
    start_time = datetime.utcnow() - timedelta(days=days)

    logs = self.query(
        user_id=user_id,
        enterprise_id=enterprise_id,
        start_time=start_time
    )

    # 统计分析
    total_actions = len(logs)
    successful_actions = sum(1 for log in logs if log.success)
    failed_actions = total_actions - successful_actions

    # 操作类型分布
    action_breakdown = {}
    for log in logs:
        action = log.action
        action_breakdown[action] = action_breakdown.get(action, 0) + 1

    return {
        "user_id": user_id,
        "period_days": days,
        "total_actions": total_actions,
        "successful_actions": successful_actions,
        "failed_actions": failed_actions,
        "success_rate": successful_actions / total_actions if total_actions > 0 else 0,
        "action_breakdown": action_breakdown,
        "first_activity": logs[-1].timestamp if logs else None,
        "last_activity": logs[0].timestamp if logs else None
    }
```

## 13.6 安全告警

### 13.6.1 异常检测

**产品化思维**：主动发现安全威胁

```python
def get_security_alerts(
    self,
    enterprise_id: Optional[str] = None,
    hours: int = 24,
    min_severity: AuditSeverity = AuditSeverity.WARNING
) -> List[AuditLog]:
    """获取安全告警"""

    # 定义安全相关的操作类型
    security_actions = [
        AuditAction.SECURITY_BLOCK.value,
        AuditAction.RATE_LIMIT_EXCEEDED.value,
        AuditAction.PERMISSION_DENIED.value,
        AuditAction.AUTH_FAILED.value,
        AuditAction.LOGIN_FAILED.value
    ]

    # 查询最近N小时的告警
    start_time = datetime.utcnow() - timedelta(hours=hours)

    alerts = []
    for action in security_actions:
        result = self.query(
            enterprise_id=enterprise_id,
            action=action,
            start_time=start_time,
            severity=min_severity.value
        )
        alerts.extend(result)

    return sorted(alerts, key=lambda x: x.timestamp, reverse=True)
```

```python
class RealTimeAlertMonitor:
    """实时告警监控"""

    def check_login_anomaly(self, enterprise_id: str):
        """检测登录异常"""
        # 1. 短时间内多次登录失败
        failed_logins = self.get_failed_logins(enterprise_id, last_1h=1)

        if failed_logins >= 5:
            self.send_alert(
                enterprise_id=enterprise_id,
                alert_type="brute_force",
                severity=AuditSeverity.CRITICAL,
                message=f"检测到暴力破解尝试：{failed_logins}次登录失败"
            )

        # 2. 异地登录
        normal_location = self.get_normal_login_location(enterprise_id)
        current_login = self.get_latest_login(enterprise_id)

        if current_login.location != normal_location:
            self.send_alert(
                enterprise_id=enterprise_id,
                alert_type="异地登录",
                severity=AuditSeverity.WARNING,
                message=f"检测到异地登录：{current_login.location}"
            )

    def check_data_access_anomaly(self, user_id: str, enterprise_id: str):
        """检测数据访问异常"""
        # 短时间内大量查询
        query_count = self.get_query_count(user_id, last_5min=5)

        if query_count >= 100:
            self.send_alert(
                enterprise_id=enterprise_id,
                user_id=user_id,
                alert_type="数据爬取",
                severity=AuditSeverity.WARNING,
                message=f"检测到异常大量查询：{query_count}次/5分钟"
            )
```
