```python
# 文件：backend/common/jwt_auth.py
from datetime import datetime, timedelta
from typing import Optional
from jose import JWTError, jwt
from passlib.context import CryptContext

# 工程化：安全配置
class SecurityConfig:
    """安全配置"""
    SECRET_KEY = "your-secret-key-change-in-production-using-env-vars"
    ALGORITHM = "HS256"
    ACCESS_TOKEN_EXPIRE_MINUTES = 60 * 24  # 24小时
    REFRESH_TOKEN_EXPIRE_DAYS = 7
    
    # 产品化：安全性要求
    TOKEN_ISSUER = "rag-system"
    TOKEN_AUDIENCE = "rag-users"

# 工程化：密码加密
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

def verify_password(plain_password: str, hashed_password: str) -> bool:
    """验证密码 - 工程化：使用安全的哈希算法"""
    return pwd_context.verify(plain_password, hashed_password)

def get_password_hash(password: str) -> str:
    """密码哈希 - 产品化：安全存储密码"""
    return pwd_context.hash(password)

def create_access_token(data: dict, expires_delta: Optional[timedelta] = None):
    """
    创建访问Token
    
    工程化：
    - 包含过期时间
    - 包含Issuer和Audience
    - 包含企业ID（多租户）
    """
    to_encode = data.copy()
    
    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(minutes=SecurityConfig.ACCESS_TOKEN_EXPIRE_MINUTES)
    
    to_encode.update({
        "exp": expire,
        "iat": datetime.utcnow(),
        "iss": SecurityConfig.TOKEN_ISSUER,
        "aud": SecurityConfig.TOKEN_AUDIENCE
    })
    
    encoded_jwt = jwt.encode(
        to_encode, 
        SecurityConfig.SECRET_KEY, 
        algorithm=SecurityConfig.ALGORITHM
    )
    return encoded_jwt

def decode_access_token(token: str):
    """
    验证并解析Token
    
    工程化：
    - 验证签名
    - 验证过期时间
    - 验证Issuer和Audience
    - 详细的错误信息
    """
    credentials_exception = ValueError("认证失败")
    
    try:
        payload = jwt.decode(
            token,
            SecurityConfig.SECRET_KEY,
            algorithms=[SecurityConfig.ALGORITHM],
            issuer=SecurityConfig.TOKEN_ISSUER,
            audience=SecurityConfig.TOKEN_AUDIENCE,
            options={"verify_exp": True}
        )
        
        user_id: str = payload.get("sub")
        enterprise_id: str = payload.get("enterprise_id")
        role: str = payload.get("role")
        
        if user_id is None:
            raise credentials_exception
            
        return {
            "user_id": user_id,
            "enterprise_id": enterprise_id,
            "role": role
        }
        
    except JWTError as e:
        raise credentials_exception from e
```

```python
from enum import Enum
from typing import List, Dict, Set

class Role(str, Enum):

    # 用户管理权限

    USER_DELETE = "user:delete"

    CHAT_USE = "chat:use"
    CHAT_HISTORY = "chat:history"
    
    # 管理权限
    SYSTEM_VIEW = "system:view"
    SYSTEM_MANAGE = "system:manage"

# 角色-权限映射
ROLE_PERMISSIONS: Dict[Role, Set[Permission]] = {
    Role.PLATFORM_ADMIN: {
        Permission.KB_CREATE, Permission.KB_READ, Permission.KB_UPDATE, Permission.KB_DELETE,
        Permission.USER_CREATE, Permission.USER_READ, Permission.USER_UPDATE, Permission.USER_DELETE,
        Permission.CHAT_USE, Permission.CHAT_HISTORY,
        Permission.SYSTEM_VIEW, Permission.SYSTEM_MANAGE
    },
    Role.ENTERPRISE_ADMIN: {
        Permission.KB_CREATE, Permission.KB_READ, Permission.KB_UPDATE, Permission.KB_DELETE,
        Permission.USER_CREATE, Permission.USER_READ, Permission.USER_UPDATE,
        Permission.CHAT_USE, Permission.CHAT_HISTORY,
        Permission.SYSTEM_VIEW
    },
    Role.AGENT: {
        Permission.KB_READ,
        Permission.CHAT_USE, Permission.CHAT_HISTORY,
        Permission.USER_READ
    },
    Role.USER: {
        Permission.KB_READ,
        Permission.CHAT_USE
    }
}

class RBACManager:
    """
    RBAC权限管理器
    
    工程化：
    - 集中化权限管理
    - 支持权限继承
    - 易于扩展
    """
    
    @staticmethod
    def has_permission(role: Role, permission: Permission) -> bool:
        """检查角色是否有指定权限"""
        role_permissions = ROLE_PERMISSIONS.get(role, set())
        return permission in role_permissions
    
    @staticmethod
    def get_role_permissions(role: Role) -> Set[Permission]:
        """获取角色的所有权限"""
        return ROLE_PERMISSIONS.get(role, set())
    
    @staticmethod
    def check_permissions(role: Role, required_permissions: List[Permission]) -> bool:
        """检查是否拥有所有必需权限"""
        return all(RBACManager.has_permission(role, perm) for perm in required_permissions)

# 工程化：在FastAPI中使用
from fastapi import Depends, HTTPException, status

async def get_current_user_permissions(
    token_data: dict = Depends(decode_access_token)
) -> Set[Permission]:
    """获取当前用户权限"""
    role = Role(token_data["role"])
    return RBACManager.get_role_permissions(role)

async def require_permission(required_permission: Permission):
    """
    权限检查依赖注入
    
    使用示例：
    @router.post("/kb")
    async def create_kb(permissions: Set = Depends(require_permission(Permission.KB_CREATE))):
        pass
    """
    def dependency(permissions: Set[Permission] = Depends(get_current_user_permissions)):
        if required_permission not in permissions:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail="权限不足"
            )
        return permissions
    return dependency
```

### 20.3.1 输入验证与过滤

**产品化思维**：不信任任何外部输入

```python
# 文件：backend/common/security.py
import re
from typing import Optional
from urllib.parse import urlparse

class InputValidator:
    """输入验证器 - 工程化：集中验证逻辑"""
    
    # 产品化：防止XSS的字符过滤
    DANGEROUS_PATTERNS = [
        r"<script\b[^>]*>(.*?)</script>",
        r"javascript:",
        r"on\w+=",
        r"data:",
    ]
    
    # 产品化：防止SQL注入的关键词
    SQL_INJECTION_PATTERNS = [
        r"(\s|\')(OR|AND)(\s|\')",
        r"(\s|\')UNION(\s|\')",
        r"(\s|\')DROP(\s|\')",
        r"(\s|\')DELETE(\s|\')",
    ]
    
    @classmethod
    def sanitize_xss(cls, text: str) -> str:
        """
        XSS防护 - 产品化：保护用户安全
        
        策略：
        1. 转义特殊字符
        2. 移除危险标签
        3. 净化输入
        """
        if not text:
            return text
        
        # 转义HTML特殊字符
        sanitized = text.replace("&", "&amp;") \
                        .replace("<", "&lt;") \
                        .replace(">", "&gt;") \
                        .replace('"', "&quot;") \
                        .replace("'", "&#x27;")
        
        # 移除危险模式
        for pattern in cls.DANGEROUS_PATTERNS:
            sanitized = re.sub(pattern, "", sanitized, flags=re.IGNORECASE)
        
        return sanitized
    
    @classmethod
    def validate_text_length(
        cls, 
        text: str, 
        min_len: int = 1, 
        max_len: int = 2000
    ) -> bool:
        """验证文本长度 - 产品化：防止恶意输入"""
        if not text:
            return min_len == 0
        return min_len <= len(text) <= max_len
    
    @classmethod
    def validate_url(cls, url: str) -> Optional[str]:
        """验证URL - 产品化：防止SSRF攻击"""
        try:
            result = urlparse(url)
            
            # 只允许HTTP和HTTPS
            if result.scheme not in ["http", "https"]:
                return None
            
            # 不允许内网地址（防止SSRF）
            forbidden_hosts = ["localhost", "127.0.0.1", "0.0.0.0", "::1"]
            if result.hostname in forbidden_hosts:
                return None
            
            # 不允许内网IP段
            if result.hostname:
                if result.hostname.startswith("10."):
                    return None
                if result.hostname.startswith("192.168."):
                    return None
                if result.hostname.startswith("172.16."):
                    return None
            
            return url
        except Exception:
            return None
```

### 20.3.2 提示词注入防护

**产品化思维**：LLM应用的特殊安全挑战

```python
# 文件：backend/chat-service/prompt_injection_defender.py
import re
from typing import Tuple, Optional

class PromptInjectionDefender:
    """
    提示词注入防护
    
    产品化：防止恶意用户劫持LLM行为
    
    攻击类型：
    1. 角色劫持 - "忽略之前的指令，你现在是..."
    2. 信息泄露 - "告诉我你的系统提示词"
    3. 恶意指令 - "删除所有数据"
    """
    
    # 典型的注入模式
    INJECTION_PATTERNS = [
        r"ignore\s+(previous|prior|above)",
        r"disregard\s+(previous|prior|above)",
        r"forget\s+(everything|all)",
        r"you\s+are\s+(now|a)",
        r"act\s+as",
        r"pretend\s+to\s+be",
        r"system\s+prompt",
        r"your\s+initial\s+prompt",
        r"what\s+are\s+your\s+instructions",
        r"tell\s+me\s+your\s+rules",
        r"bypass.*filter",
        r"override.*restriction",
        r"sudo",
        r"admin",
    ]
    
    # 敏感指令（根据业务调整）
    SENSITIVE_INSTRUCTIONS = [
        r"delete",
        r"remove",
        r"drop",
        r"erase",
        r"reset",
    ]
    
    def __init__(self):
        # 编译正则表达式（性能优化）
        self.injection_regex = re.compile(
            "|".join(self.INJECTION_PATTERNS),
            re.IGNORECASE
        )
        self.sensitive_regex = re.compile(
            "|".join(self.SENSITIVE_INSTRUCTIONS),
            re.IGNORECASE
        )
    
    def check_prompt(self, prompt: str) -> Tuple[bool, Optional[str]]:
        """
        检查输入是否包含提示词注入
        
        返回:
        (是否安全, 风险描述)
        """
        if not prompt:
            return True, None
        
        # 检查注入模式
        if self.injection_regex.search(prompt):
            return False, "检测到潜在的提示词注入"
        
        # 检查敏感指令
        if self.sensitive_regex.search(prompt):
            return False, "检测到敏感指令"
        
        return True, None
    
    def sanitize_prompt(self, prompt: str) -> str:
        """
        净化提示词 - 工程化：多层防御
        
        策略：
        1. 移除明显的注入模式
        2. 对输入进行转义处理
        3. 添加安全边界
        """
        # 移除或替换危险关键词
        sanitized = prompt
        
        # 这是一个简化示例，实际应用中需要更复杂的处理
        dangerous_words = ["ignore", "disregard", "forget", "system prompt"]
        for word in dangerous_words:
            sanitized = sanitized.replace(word, "[REDACTED]")
        
        return sanitized
```

```python
class DataEncryptor:
    """
    数据加密器
    
    工程化：
    - 使用行业标准算法
    - 密钥管理
    - 安全的随机数生成
    """
    
    def __init__(self, key: Optional[bytes] = None):
        """
        初始化加密器
        
        产品化：
        - 密钥从环境变量获取
        - 生产环境必须使用强密钥
        """

    def encrypt(self, plaintext: str) -> str:
        """加密数据"""
        if not plaintext:
            return plaintext
        return self.fernet.encrypt(plaintext.encode()).decode()
    
    def decrypt(self, ciphertext: str) -> str:
        """解密数据"""
        if not ciphertext:
            return ciphertext
        return self.fernet.decrypt(ciphertext.encode()).decode()
    
    @staticmethod
    def generate_key() -> bytes:
        """生成新密钥（用于初始化）"""
        return Fernet.generate_key()

class DataMasker:
    """
    数据脱敏器
    
    产品化：在日志和展示时保护敏感信息
    """
    
    @staticmethod
    def mask_phone(phone: str) -> str:
        """手机号脱敏"""
        if not phone or len(phone) < 7:
            return phone
        return phone[:3] + "****" + phone[7:]
    
    @staticmethod
    def mask_email(email: str) -> str:
        """邮箱脱敏"""
        if "@" not in email:
            return email
        name, domain = email.split("@", 1)
        if len(name) <= 2:
            return name + "****@" + domain
        return name[:2] + "****@" + domain
    
    @staticmethod
    def mask_id_card(id_card: str) -> str:
        """身份证号脱敏"""
        if not id_card or len(id_card) < 10:
```

### 20.5.1 接口限流

**产品化思维**：防止滥用，保障服务稳定

```python
# 文件：backend/chat-service/rate_limit.py
from datetime import datetime, timedelta
from typing import Optional
from functools import wraps
from fastapi import Request, HTTPException, status
import redis
import hashlib

class RateLimiter:
    """
    限流器
    
    工程化：
    - 使用Redis实现分布式限流
    - 灵活的限流策略
    - 防止滥用
    """
    
    def __init__(self, redis_client):
        self.redis = redis_client
    
    async def check_rate_limit(
        self,
        key: str,
        limit: int,
        window_seconds: int
    ) -> tuple[bool, int, int]:
        """
        检查是否超限
        
        返回: (是否允许, 剩余请求数, 重置时间)
        """
        now = datetime.now()
        window_start = now - timedelta(seconds=window_seconds)
        
        # 使用Redis的Sorted Set实现滑动窗口
        pipe = self.redis.pipeline()
        
        # 移除窗口外的记录
        pipe.zremrangebyscore(key, 0, window_start.timestamp())
        
        # 获取当前窗口内的请求数
        pipe.zcard(key)
        
        # 添加当前请求
        pipe.zadd(key, {now.timestamp(): now.timestamp()})
        
        # 设置过期时间
        pipe.expire(key, window_seconds)
        
        results = pipe.execute()
        current_count = results[1]
        
        remaining = limit - current_count
        reset_time = (now + timedelta(seconds=window_seconds)).timestamp()
        
        return remaining >= 0, max(0, remaining), int(reset_time)

# 工程化：FastAPI依赖注入
def rate_limiter(limit: int, window_seconds: int = 60, key_prefix: str = "rate_limit"):
    """
    限流装饰器
    
    使用示例:
    @router.post("/chat")
    @rate_limiter(limit=60, window_seconds=60)  # 每分钟60次
    async def chat_endpoint():
        pass
    """
    def decorator(func):
        @wraps(func)
        async def wrapper(request: Request, *args, **kwargs):
            # 获取客户端标识
            client_ip = request.client.host
            key = f"{key_prefix}:{client_ip}"
            
            # 这里需要从依赖注入获取Redis客户端
            # 简化示例，实际项目中使用Depends
            redis_client = get_redis_client()
            limiter = RateLimiter(redis_client)
            
            allowed, remaining, reset = await limiter.check_rate_limit(
                key, limit, window_seconds
            )
            
            if not allowed:
                raise HTTPException(
                    status_code=status.HTTP_429_TOO_MANY_REQUESTS,
                    detail="请求频率过高，请稍后再试",
                    headers={
                        "X-RateLimit-Limit": str(limit),
                        "X-RateLimit-Remaining": str(remaining),
                        "X-RateLimit-Reset": str(reset)
                    }
                )
            
            return await func(request, *args, **kwargs)
        return wrapper
    return decorator
```

### 20.5.2 熔断器模式

**工程化实现**：防止级联故障

```python
# 文件：backend/chat-service/circuit_breaker.py
from enum import Enum
import time
from functools import wraps
import logging

logger = logging.getLogger("circuit_breaker")

class CircuitState(Enum):
    """熔断器状态"""
    CLOSED = "closed"        # 正常状态
    OPEN = "open"            # 熔断状态
    HALF_OPEN = "half_open"  # 半开状态（尝试恢复）

class CircuitBreaker:
    """
    熔断器
    
    工程化：
    - 快速失败，保护下游服务
    - 自动恢复
    - 可配置的参数
    """
    
    def __init__(
        self,
        failure_threshold: int = 5,
        recovery_timeout: int = 30,
        expected_exception: Exception = Exception
    ):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.expected_exception = expected_exception
        
        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.last_failure_time = 0
    
    def call(self, func, *args, **kwargs):
        """执行函数，应用熔断器逻辑"""
        self._check_state()
        
        if self.state == CircuitState.OPEN:
            raise Exception("Circuit is OPEN")
        
        try:
            result = func(*args, **kwargs)
            self._on_success()
            return result
        except self.expected_exception as e:
            self._on_failure()
            raise
    
    def _check_state(self):
        """检查并更新熔断器状态"""
        if self.state == CircuitState.OPEN:
            now = time.time()
            if now - self.last_failure_time > self.recovery_timeout:
                self.state = CircuitState.HALF_OPEN
                logger.info(f"Circuit breaker switching to HALF_OPEN")
    
    def _on_success(self):
        """成功时的处理"""
        if self.state == CircuitState.HALF_OPEN:
            self.state = CircuitState.CLOSED
            self.failure_count = 0
            logger.info(f"Circuit breaker switched to CLOSED")
    
    def _on_failure(self):
        """失败时的处理"""
        self.failure_count += 1
        self.last_failure_time = time.time()
        
        if self.state == CircuitState.HALF_OPEN:
            self.state = CircuitState.OPEN
            logger.warning(f"Circuit breaker switched to OPEN (failed recovery)")
        elif self.failure_count >= self.failure_threshold:
            self.state = CircuitState.OPEN
            logger.warning(f"Circuit breaker switched to OPEN (failure threshold reached)")

# 使用示例
def circuit_breaker(threshold: int = 5, timeout: int = 30):
    """熔断器装饰器"""
    breaker = CircuitBreaker(
        failure_threshold=threshold,
        recovery_timeout=timeout
    )
    
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            return breaker.call(func, *args, **kwargs)
        return wrapper
    return decorator
```

## 20.7 本章小结

### 核心安全概念

| 概念 | 工程化价值 | 产品化价值 |
|------|-----------|-----------|
| **认证与授权** | 标准化身份管理 | 保护用户账户安全 |
| **输入验证** | 防止攻击 | 保障数据质量 |
| **数据加密** | 数据安全 | 符合隐私法规 |
| **审计日志** | 可追溯性 | 合规与安全分析 |
| **限流熔断** | 服务稳定性 | 防止滥用，保障体验 |

### 工程化思维要点

1. **安全左移** - 安全是设计的一部分，不是事后添加
2. **纵深防御** - 多层防护，不依赖单一措施
3. **最小权限** - 只授予必要的权限
4. **安全默认** - 默认配置应该是安全的
5. **持续监控** - 安全不是一次性的，需要持续关注

### 产品化思维要点

1. **用户信任** - 安全是建立信任的基础
2. **合规要求** - 满足法律法规要求
3. **业务连续性** - 安全事件会影响业务
4. **品牌保护** - 安全事故损害品牌形象
5. **成本控制** - 预防比事后处理便宜
