## 3. 各层级缓存详解

### 27.1 FAQ Cache - 第一层缓存

```python
def _build_faq_key(self, query, enterprise_id):
    """
    FAQ 缓存 key 构建
    
    设计思路：
    1. 规范化查询：转小写、去空格
    2. Hash 计算：MD5 前16位
    3. Key 格式：faq:{enterprise_id}:{hash}
    """
    query_hash = hashlib.md5(query.lower().strip().encode()).hexdigest()[:16]
    return f"{self.faq_cache_prefix}{enterprise_id}:{query_hash}"

async def get_faq_cache(self, query, enterprise_id):
    """获取 FAQ 缓存"""
    key = self._build_faq_key(query, enterprise_id)
    result = await self.redis.get(key)
    return result.decode('utf-8') if result else None

async def set_faq_cache(self, query, enterprise_id, answer):
    """设置 FAQ 缓存"""
    key = self._build_faq_key(query, enterprise_id)
    await self.redis.set(key, answer, ex=self.faq_cache_ttl)
    return True
```

### 27.2 Retrieval Cache - 第二层缓存

```python
def _build_retrieval_key(
    self, query, enterprise_id, knowledge_base_ids=None
):
    """
    检索缓存 key 构建
    
    设计考虑：
    1. 需要包含知识库 IDs（因为不同知识库结果不同）
    2. 知识库 IDs 排序后再 Hash（顺序不同但内容相同）
    """
    query_hash = hashlib.md5(query.lower().strip().encode()).hexdigest()[:16]
    
    kb_hash = ""
    if knowledge_base_ids:
        kb_str = "-".join(sorted(knowledge_base_ids))
        kb_hash = f":{hashlib.md5(kb_str.encode()).hexdigest()[:8]}"
    
    return f"{self.retrieval_cache_prefix}{enterprise_id}:{query_hash}{kb_hash}"

async def get_retrieval_cache(self, query, enterprise_id, knowledge_base_ids=None):
    """获取检索缓存"""
    key = self._build_retrieval_key(query, enterprise_id, knowledge_base_ids)
    result = await self.redis.get(key)
    if result:
        return json.loads(result.decode('utf-8'))
    return None

async def set_retrieval_cache(self, query, enterprise_id, chunks, knowledge_base_ids=None):
    """设置检索缓存"""
    key = self._build_retrieval_key(query, enterprise_id, knowledge_base_ids)
    await self.redis.set(key, json.dumps(chunks), ex=self.retrieval_cache_ttl)
    return True
```

## 4. 缓存失效策略

### 27.1 主动失效

```python
async def invalidate_faq_cache(self, enterprise_id=None):
    """
    失效 FAQ 缓存
    
    使用场景：
    - 知识库更新后
    - FAQ 配置变更
    """
    pattern = f"{self.faq_cache_prefix}*" if enterprise_id is None \
        else f"{self.faq_cache_prefix}{enterprise_id}:*"
    
    keys = await self.redis.keys(pattern)
    if keys:
        await self.redis.delete(*keys)
    return True

async def invalidate_retrieval_cache(self, enterprise_id=None):
    """失效检索缓存"""
    pattern = f"{self.retrieval_cache_prefix}*" if enterprise_id is None \
        else f"{self.retrieval_cache_prefix}{enterprise_id}:*"
    
    keys = await self.redis.keys(pattern)
    if keys:
        await self.redis.delete(*keys)
    return True
```

### 27.2 被动失效（TTL）

```
TTL 设计原则：

FAQ Cache → 24小时（数据相对稳定）
Retrieval Cache → 1小时（可能有新增内容）
Session Cache → 30分钟（会话活跃时间短）
```

async def rag_pipeline(enterprise_id, query, knowledge_base_ids):
    """
    完整的 RAG 流程，集成缓存
    """

    # 第1步：FAQ 缓存检查

    if is_simple_qa(query):

    return {
        "source": "rag",
        "answer": answer,
        "chunks": chunks
    }

## 6. 工程化思维要点

### 27.1 Key 设计最佳实践

```
Key 设计原则：

✅ 好的设计：
- 包含租户/企业 ID（多租户）
- 包含必要的参数
- 使用 Hash 缩短长度
- 命名清晰，容易理解

❌ 不好的设计：
- 太长（Redis Key 限制 512MB）
- 没有前缀（难管理）
- 包含敏感信息

示例：
faq:{enterprise_id}:{query_hash} ✅
retrieval:{enterprise_id}:{query_hash}:{kb_hash} ✅
mykey123 ❌
```

### 27.2 缓存雪崩、击穿、穿透

```python
# 缓存雪崩：同一时间大量 key 过期
# 解决方案：TTL 加随机值
import random

def get_ttl_with_jitter(base_ttl):
    return base_ttl + random.randint(-300, 300)  # ±5分钟

# 缓存击穿：热点 key 过期时大量请求
# 解决方案：互斥锁（只让一个请求去更新）
async def get_with_lock(key, fetch_func):
    lock_key = f"lock:{key}"
    
    # 尝试获取锁
    if await redis.set(lock_key, "1", nx=True, ex=10):
        try:
            # 重新查询缓存（可能其他请求刚设置）
            data = await redis.get(key)
            if data:
                return data
            
            # 获取数据
            data = await fetch_func()
            await redis.set(key, data, ex=TTL)
            return data
        finally:
            await redis.delete(lock_key)
    else:
        # 等待并重试
        await asyncio.sleep(0.1)
        return await get_with_lock(key, fetch_func)

# 缓存穿透：查询不存在的数据
# 解决方案：缓存空值
async def get_with_null_caching(key, fetch_func):
    data = await redis.get(key)
    
    if data == b"NULL":  # 空值标记
        return None
    
    if data:
        return data
    
    # 实际查询
    result = await fetch_func()
    
    if result is None:
        await redis.set(key, "NULL", ex=300)  # 缓存空值5分钟
    else:
        await redis.set(key, result, ex=TTL)
    
    return result
```

## 7. 产品化思维要点

### 27.1 成本优化

```
成本分析：

假设：
- LLM API 调用 $0.01 / 次
- FAQ 缓存命中率 60%
- Retrieval 缓存命中率 30%
- 每天 10,000 次请求

没有缓存：
$0.01 × 10,000 = $100 / 天

有缓存：
FAQ 命中 6,000 → 节省 $60
Retrieval 命中 3,000 → 节省检索成本
实际调用 1,000 → $10 / 天

节省：90% 成本 🎉
```

| LLM 调用量 | 100% | 10-30% |
