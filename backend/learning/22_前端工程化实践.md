## 22.2 状态管理

### 22.2.1 Zustand 实践

**工程化实践**：轻量且灵活的状态管理

```typescript
// 文件：frontend/packages/store/chatStore.ts
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';
import type { ChatMessage, ChatSession } from '@repo/api-client';

interface ChatState {
  // 状态
  currentSessionId: string | null;
  sessions: ChatSession[];
  messages: Record<string, ChatMessage[]>; // sessionId -> messages
  isLoading: boolean;
  error: string | null;
  
  // 操作
  setCurrentSession: (sessionId: string | null) => void;
  addSession: (session: ChatSession) => void;
  addMessage: (sessionId: string, message: ChatMessage) => void;
  setLoading: (loading: boolean) => void;
  setError: (error: string | null) => void;
  clearError: () => void;
}

export const useChatStore = create<ChatState>()(
  // 工程化：使用persist中间件持久化状态
  persist(
    (set, get) => ({
      currentSessionId: null,
      sessions: [],
      messages: {},
      isLoading: false,
      error: null,
      
      setCurrentSession: (sessionId) => 
        set({ currentSessionId: sessionId }),
      
      addSession: (session) =>
        set((state) => ({ 
          sessions: [...state.sessions, session] 
        })),
      
      addMessage: (sessionId, message) =>
        set((state) => ({
          messages: {
            ...state.messages,
            [sessionId]: [...(state.messages[sessionId] || []), message]
          }
        })),
      
      setLoading: (loading) => set({ isLoading: loading }),
      setError: (error) => set({ error }),
      clearError: () => set({ error: null }),
    }),
    {
      name: 'chat-storage', // 存储名称
      storage: createJSONStorage(() => localStorage),
      // 产品化：只持久化必要的状态
      partialize: (state) => ({
        sessions: state.sessions,
        currentSessionId: state.currentSessionId,
      }),
    }
  )
);

// 在组件中使用
function ChatComponent() {
  const { currentSessionId, messages, setCurrentSession } = useChatStore();
  // ...
}
```

```typescript
// 文件：frontend/packages/store/selectors.ts
import { useShallow } from 'zustand/react/shallow';
import { useChatStore } from './chatStore';

// 工程化：使用shallow比较优化性能
export function useCurrentMessages() {
  return useChatStore(
    useShallow((state) => {
      const sessionId = state.currentSessionId;
      return sessionId ? state.messages[sessionId] || [] : [];
    })
  );
}

// 或者使用memoized selectors
import { create } from 'zustand';
import { createSelector } from 'reselect'; // 或者用zustand自带的

const selectChatState = (state: ChatState) => state;

export const selectCurrentMessages = createSelector(
  [selectChatState],
  (state) => {
    const sessionId = state.currentSessionId;
    return sessionId ? state.messages[sessionId] || [] : [];
  }
);
```

## 22.3 API 客户端设计

### 22.3.1 类型安全的 API 客户端

**工程化实践**：与后端类型保持一致

```typescript
// 文件：frontend/packages/api-client/index.ts
import axios, { AxiosInstance, AxiosError } from 'axios';
import type { 
  ChatRequest, 
  ChatResponse, 
  KnowledgeBase,
  ApiResponse 
} from './types';

// 工程化：统一的错误处理
class ApiError extends Error {
  constructor(
    message: string,
    public status?: number,
    public code?: string
  ) {
    super(message);
    this.name = 'ApiError';
  }
}

// 工程化：API客户端类
class APIClient {
  private client: AxiosInstance;
  private token: string | null = null;
  
  constructor(baseURL: string = '/api/v1') {
    this.client = axios.create({
      baseURL,
      timeout: 30000,
      headers: {
        'Content-Type': 'application/json',
      },
    });
    
    // 工程化：请求拦截器
    this.client.interceptors.request.use(
      (config) => {
        if (this.token) {
          config.headers.Authorization = `Bearer ${this.token}`;
        }
        return config;
      },
      (error) => Promise.reject(error)
    );
    
    // 工程化：响应拦截器 - 统一错误处理
    this.client.interceptors.response.use(
      (response) => response.data,
      (error: AxiosError) => {
        const apiError = new ApiError(
          error.response?.data?.message || error.message,
          error.response?.status
        );
        return Promise.reject(apiError);
      }
    );
  }
  
  setToken(token: string) {
    this.token = token;
  }
  
  clearToken() {
    this.token = null;
  }
  
  // Chat API
  async createSession(): Promise<ApiResponse<{ sessionId: string }>> {
    return this.client.post('/chat/sessions');
  }
  
  async sendMessage(
    sessionId: string, 
    data: ChatRequest
  ): Promise<ApiResponse<ChatResponse>> {
    return this.client.post(`/chat/sessions/${sessionId}/messages`, data);
  }
  
  // Knowledge API
  async getKnowledgeBases(): Promise<ApiResponse<KnowledgeBase[]>> {
    return this.client.get('/knowledge-bases');
  }
  
  // 产品化：流式请求 - SSE
  streamMessage(
    sessionId: string, 
    query: string
  ): EventSource {
    const url = new URL(`/chat/sessions/${sessionId}/messages/stream`, this.client.defaults.baseURL);
    url.searchParams.set('query', query);
    
    const eventSource = new EventSource(url.toString(), {
      withCredentials: true,
    });
    
    return eventSource;
  }
}

// 单例实例
export const apiClient = new APIClient(import.meta.env.VITE_API_URL || '/api/v1');
```

### 22.3.2 React Query 数据管理

**产品化实践**：缓存、重试、乐观更新

```typescript
// 文件：frontend/apps/consumer/src/hooks/useChat.ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { apiClient } from '@repo/api-client';
import { useChatStore } from '@repo/store';

export function useChatSessions() {
  return useQuery({
    queryKey: ['chatSessions'],
    queryFn: () => apiClient.getChatSessions(),
    // 产品化：5分钟缓存
    staleTime: 1000 * 60 * 5,
  });
}

export function useSendMessage(sessionId: string) {
  const queryClient = useQueryClient();
  const addMessage = useChatStore((state) => state.addMessage);
  
  return useMutation({
    mutationFn: (data: { query: string; stream?: boolean }) =>
      apiClient.sendMessage(sessionId, data),
    
    // 产品化：乐观更新 - 先显示用户消息
    onMutate: async (data) => {
      // 1. 取消正在进行的请求
      await queryClient.cancelQueries({ queryKey: ['messages', sessionId] });
      
      // 2. 添加用户消息到本地
      const tempMessage = {
        id: `temp-${Date.now()}`,
        role: 'user' as const,
        content: data.query,
        timestamp: new Date().toISOString(),
      };
      addMessage(sessionId, tempMessage);
      
      return { tempMessage };
    },
    
    // 产品化：成功后更新数据
    onSuccess: (response) => {
      addMessage(sessionId, response.data);
      // 使会话列表失效，刷新
      queryClient.invalidateQueries({ queryKey: ['chatSessions'] });
    },
    
    // 产品化：失败回滚
    onError: (error, _, context) => {
      if (context?.tempMessage) {
        // 回滚乐观更新
        // ...
      }
    },
    
    // 工程化：失败重试
    retry: 2,
    retryDelay: (attempt) => Math.min(1000 * 2 ** attempt, 30000),
  });
}
```

```typescript
import React from 'react';
import { cva, type VariantProps } from 'class-variance-authority';
import { clsx } from 'clsx';

// 工程化：使用CVA管理组件变体
const buttonVariants = cva(
  'inline-flex items-center justify-center rounded-md font-medium transition-colors focus:outline-none focus:ring-2',
  {
    variants: {
      variant: {
        primary: 'bg-blue-600 text-white hover:bg-blue-700 focus:ring-blue-500',
        secondary: 'bg-gray-100 text-gray-900 hover:bg-gray-200 focus:ring-gray-500',
        danger: 'bg-red-600 text-white hover:bg-red-700 focus:ring-red-500',
        ghost: 'hover:bg-gray-100 text-gray-700',
      },
      size: {
        sm: 'px-3 py-1.5 text-sm',
        md: 'px-4 py-2 text-base',
        lg: 'px-6 py-3 text-lg',
      },
      fullWidth: {
        true: 'w-full',
        false: '',
      },
    },
    defaultVariants: {
      variant: 'primary',
      size: 'md',
      fullWidth: false,
    },
  }
);

interface ButtonProps 
  extends React.ButtonHTMLAttributes<HTMLButtonElement>,
    VariantProps<typeof buttonVariants> {
  // 产品化：加载状态
  loading?: boolean;
  // 产品化：图标
  leftIcon?: React.ReactNode;
  rightIcon?: React.ReactNode;
}

export const Button = React.forwardRef<HTMLButtonElement, ButtonProps>(
  (
    {
      className,
      variant,
      size,
      fullWidth,
      loading,
      leftIcon,
      rightIcon,
      disabled,
      children,
      ...props
    },
    ref
  ) => {
    return (
      <button
        ref={ref}
        className={clsx(buttonVariants({ variant, size, fullWidth }), className)}
        disabled={disabled || loading}
        {...props}
      >
        {/* 产品化：加载动画 */}
        {loading && (
          <svg 
            className="animate-spin -ml-1 mr-2 h-4 w-4" 
            viewBox="0 0 24 24"
          >
            <circle cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4" fill="none" strokeDasharray="150" strokeDashoffset="50" />
          </svg>
        )}
        
        {leftIcon && <span className="mr-2">{leftIcon}</span>}
        {children}
        {rightIcon && <span className="ml-2">{rightIcon}</span>}
      </button>
    );
  }
);

Button.displayName = 'Button';
```

```typescript
export function LazyImage({ src, alt, ...props }: React.ImgHTMLAttributes<HTMLImageElement>) {
  const [isLoaded, setIsLoaded] = React.useState(false);
  
  return (
    <div className="relative">
      {/* 占位图 */}
      {!isLoaded && (
        <div className="absolute inset-0 bg-gray-200 animate-pulse" />
      )}

        src={src}
        alt={alt}
        loading="lazy"
        onLoad={() => setIsLoaded(true)}
        className={`transition-opacity duration-300 ${isLoaded ? 'opacity-100' : 'opacity-0'}`}
        {...props}
      />
    </div>
  );
}
```

## 22.8 本章小结

### 核心前端概念

| 概念 | 工程化价值 | 产品化价值 |
|------|-----------|-----------|
| **TypeScript** | 类型安全，减少bug | 稳定可靠 |
| **状态管理** | 数据流清晰 | 用户体验连贯 |
| **组件设计** | 可复用，可维护 | 一致的UI/UX |
| **性能优化** | 加载快，渲染快 | 用户体验流畅 |
| **测试** | 代码质量保障 | 产品稳定 |

### 工程化思维要点

1. **模块化** - 组件、状态、API分离
2. **类型安全** - TypeScript全流程
3. **可测试性** - 组件易于测试
4. **构建优化** - 代码分割、懒加载
5. **统一工具链** - Monorepo + 统一配置

### 产品化思维要点

1. **用户体验** - 性能优化、加载状态
2. **错误处理** - 友好的错误提示
3. **交互反馈** - 加载、成功、失败状态
4. **可访问性** - A11y支持
5. **性能预算** - 设置性能指标并监控
