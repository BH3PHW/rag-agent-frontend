## 19.1 数据清洗的重要性

### 19.1.1 垃圾数据的危害

```
垃圾数据的后果：
┌──────────────────────────────────────────────────────────────┐
│  用户问："产品退货政策是什么？"                               │
│  ↓                                                         │
│  RAG检索到："客服电话：123456"  "2023年放假通知"             │
│  ↓                                                         │
│  LLM基于垃圾信息回答："客服电话是123456，2023年放假安排..."   │
│  ↓                                                         │
│  用户体验差 → 用户流失 → 产品失败                             │
└──────────────────────────────────────────────────────────────┘
```

### 19.1.2 数据清洗的价值

| 清洗环节 | 解决的问题 | 产品收益 |
|---------|----------|---------|
| 格式规范化 | 解析错误、乱码 | 数据可用率提升30% |
| 去重处理 | 冗余信息、检索干扰 | 检索效率提升50% |
| 内容过滤 | 敏感信息、低质量内容 | 安全性、准确性提升 |
| 结构化处理 | 无序文本 | 检索精度提升20%+ |

## 19.2 数据清洗流程详解

### 19.2.1 完整清洗管道

```

```

### 19.2.2 代码实现

```python
# 文件：backend/knowledge-service/data_cleaner.py
import re
from typing import List, Dict, Tuple
import logging

logger = logging.getLogger("data-cleaner")

class DataCleaner:
    """
    数据清洗器 - 工程化：模块化清洗管道
    
    设计思路：
    - 链式调用，可组合
    - 每个清洗步骤可配置
    - 清洗记录可观测
    """
    
    def __init__(self):
        self.cleaning_stats = {
            "total_documents": 0,
            "cleaned_documents": 0,
            "filtered_out": 0,
            "issues_found": 0
        }
    
    def clean_document(self, content: str, metadata: Dict) -> Tuple[str, Dict]:
        """
        完整清洗流程
        
        Args:
            content: 原始文本内容
            metadata: 元数据
            
        Returns:
            (清洗后的内容, 清洗统计)
        """
        self.cleaning_stats["total_documents"] += 1
        stats = {"issues": []}
        
        # 阶段1：基础清洗
        content, issue1 = self._basic_cleanup(content)
        if issue1:
            stats["issues"].extend(issue1)
        
        # 阶段2：敏感内容过滤
        content, issue2 = self._filter_sensitive_content(content)
        if issue2:
            stats["issues"].extend(issue2)
        
        # 阶段3：去除无意义内容
        content, issue3 = self._remove_noise(content)
        if issue3:
            stats["issues"].extend(issue3)
        
        # 阶段4：质量检查
        if not self._is_quality_content(content):
            self.cleaning_stats["filtered_out"] += 1
            raise ValueError("文档质量不达标")
        
        self.cleaning_stats["cleaned_documents"] += 1
        self.cleaning_stats["issues_found"] += len(stats["issues"])
        
        return content, stats
    
    def _basic_cleanup(self, text: str) -> Tuple[str, List[str]]:
        """基础清洗"""
        issues = []
        original_length = len(text)
        
        # 1. 统一换行符
        text = text.replace("\r\n", "\n").replace("\r", "\n")
        
        # 2. 去除多余空白
        # 工程化：保留段落语义
        text = re.sub(r'[ \t]+', ' ', text)
        text = re.sub(r'\n{3,}', '\n\n', text)
        
        # 3. 去除不可见字符
        text = re.sub(r'[\x00-\x08\x0b\x0c\x0e-\x1f\x7f]', '', text)
        
        # 4. 去除控制字符
        if len(text) != original_length:
            issues.append("去除了不可见字符")
        
        return text.strip(), issues
    
    def _filter_sensitive_content(self, text: str) -> Tuple[str, List[str]]:
        """
        敏感内容过滤 - 产品化：安全保障
        
        涵盖：
        - 个人隐私信息
        - 商业敏感信息
        - 不适当内容
        """
        issues = []
        
        # 1. 手机号脱敏
        text, count = re.subn(r'1[3-9]\d{9}', '[手机号]', text)
        if count > 0:
            issues.append(f"脱敏了{count}个手机号")
        
        # 2. 邮箱脱敏
        text, count = re.subn(r'[\w.-]+@[\w.-]+', '[邮箱]', text)
        if count > 0:
            issues.append(f"脱敏了{count}个邮箱")
        
        # 3. 身份证号脱敏
        text, count = re.subn(r'\d{17}[\dXx]', '[身份证号]', text)
        if count > 0:
            issues.append(f"脱敏了{count}个身份证号")
        
        # 4. 敏感词过滤
        sensitive_words = ["敏感词1", "敏感词2", "敏感词3"]
        for word in sensitive_words:
            if word in text:
                text = text.replace(word, '***')
                issues.append(f"过滤了敏感词: {word}")
        
        return text, issues
    
    def _remove_noise(self, text: str) -> Tuple[str, List[str]]:
        """
        去除噪音内容 - 工程化：提升检索质量
        
        噪音包括：
        - 重复内容
        - 无意义装饰
        - 页脚页眉
        """
        issues = []
        
        # 1. 去除重复段落
        lines = text.split('\n')
        seen = set()
        unique_lines = []
        removed = 0
        
        for line in lines:
            line_stripped = line.strip()
            if len(line_stripped) > 20:  # 过长行哈希去重
                line_hash = hash(line_stripped)
                if line_hash in seen:
                    removed += 1
                    continue
                seen.add(line_hash)
            unique_lines.append(line)
        
        if removed > 0:
            issues.append(f"去除了{removed}个重复段落")
        
        text = '\n'.join(unique_lines)
        
        # 2. 去除过短文档
        if len(text) < 50:
            issues.append("文档内容过短")
        
        # 3. 去除特殊垃圾
        # 例如："Page 1 of 10"
        text = re.sub(r'Page \d+ of \d+', '', text, flags=re.IGNORECASE)
        
        return text, issues
    
    def _is_quality_content(self, text: str) -> bool:
        """
        质量检查 - 产品化：保障知识库质量
        
        检查项：
        - 文本长度
        - 有效字符比例
        - 可读性
        """
        # 长度检查
        if len(text) < 50:
            return False
        
        # 有效字符比例（过多符号说明可能是垃圾）
        letter_ratio = sum(c.isalpha() or c.isdigit() for c in text) / len(text)
        if letter_ratio < 0.3:
            return False
        
        return True
    
    def get_cleaning_report(self) -> Dict:
        """获取清洗报告 - 工程化：可观测性"""
        return {
            "total_processed": self.cleaning_stats["total_documents"],
            "successfully_cleaned": self.cleaning_stats["cleaned_documents"],
            "filtered_out": self.cleaning_stats["filtered_out"],
            "total_issues_found": self.cleaning_stats["issues_found"],
            "cleaning_rate": self.cleaning_stats["cleaned_documents"] / 
                           max(self.cleaning_stats["total_documents"], 1)
        }
```

## 19.3 文档分块优化 - 影响检索质量的关键

### 19.3.1 分块策略

```python
# 文件：backend/knowledge-service/chunker.py
from typing import List, Dict
import re

class IntelligentChunker:
    """
    智能文档分块器 - 工程化：保障检索质量
    
    设计思路：
    - 保持语义完整性
    - 考虑上下文关联
    - 可配置策略
    """
    
    def __init__(
        self,
        chunk_size: int = 512,      # 分块大小
        chunk_overlap: int = 100     # 重叠大小
    ):
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
    
    def chunk_document(
        self,
        text: str,
        metadata: Dict = None
    ) -> List[Dict]:
        """智能分块"""
        
        # 步骤1：文档结构化分析
        structure = self._analyze_document_structure(text)
        
        # 步骤2：按结构分块
        chunks = []
        
        if structure.get("has_sections"):
            chunks = self._chunk_by_section(text, structure)
        else:
            chunks = self._chunk_by_paragraph(text)
        
        # 步骤3：添加元数据
        for i, chunk in enumerate(chunks):
            chunk["metadata"] = {
                **(metadata or {}),
                "chunk_index": i,
                "total_chunks": len(chunks),
                "chunk_size": len(chunk["text"])
            }
        
        return chunks
    
    def _analyze_document_structure(self, text: str) -> Dict:
        """分析文档结构"""
        lines = text.split('\n')
        
        structure = {
            "has_sections": False,
            "sections": [],
            "has_lists": False,
            "avg_paragraph_length": 0
        }
        
        # 检测标题
        title_pattern = r'^#{1,6}\s+.+|^[A-Z][A-Z\s]+$'
        
        for line in lines:
            line = line.strip()
            if re.match(title_pattern, line):
                structure["has_sections"] = True
                structure["sections"].append(line)
        
        return structure
    
    def _chunk_by_section(self, text: str, structure: Dict) -> List[Dict]:
        """按章节分块 - 保持语义完整性"""
        chunks = []
        
        # 简单实现：每个章节一个块
        sections = re.split(r'^#{1,6}\s', text, flags=re.MULTILINE)
        
        current_chunk = ""
        for section in sections:
            if len(current_chunk) + len(section) < self.chunk_size:
                current_chunk += "\n" + section
            else:
                if current_chunk.strip():
                    chunks.append({"text": current_chunk.strip()})
                current_chunk = section
        
        if current_chunk.strip():
            chunks.append({"text": current_chunk.strip()})
        
        return chunks
    
    def _chunk_by_paragraph(self, text: str) -> List[Dict]:
        """按段落分块"""
        paragraphs = re.split(r'\n{2,}', text)
        
        chunks = []
        current_chunk = ""
        
        for para in paragraphs:
            if len(current_chunk) + len(para) < self.chunk_size:
                current_chunk += "\n\n" + para
            else:
                if current_chunk.strip():
                    chunks.append({"text": current_chunk.strip()})
                
                # 段落太长，需要拆分
                if len(para) > self.chunk_size:
                    # 重叠滑动窗口
                    for i in range(0, len(para), self.chunk_size - self.chunk_overlap):
                        chunk_text = para[i:i + self.chunk_size]
                        chunks.append({"text": chunk_text})
                    current_chunk = ""
                else:
                    current_chunk = para
        
        if current_chunk.strip():
            chunks.append({"text": current_chunk.strip()})
        
        return chunks
```

## 19.4 数据清洗监控与质量评估

### 19.4.1 质量评估指标

```python
# 文件：backend/knowledge-service/quality_evaluator.py
from typing import List, Dict
import logging

logger = logging.getLogger("quality-evaluator")

class QualityEvaluator:
    """
    知识库质量评估 - 工程化：可观测性
    
    定期评估知识库质量，及时发现问题
    """
    
    @classmethod
    def evaluate_knowledge_base(cls, chunks: List[Dict]) -> Dict:
        """评估知识库质量"""
        evaluation = {
            "total_chunks": len(chunks),
            "avg_chunk_length": 0,
            "quality_score": 0,
            "issues": []
        }
        
        if not chunks:
            return evaluation
        
        # 计算平均块长度
        total_length = sum(len(chunk.get("text", "")) for chunk in chunks)
        evaluation["avg_chunk_length"] = total_length / len(chunks)
        
        # 检查块长度分布
        too_short = sum(1 for chunk in chunks if len(chunk.get("text", "")) < 100)
        too_long = sum(1 for chunk in chunks if len(chunk.get("text", "")) > 2000)
        
        if too_short > len(chunks) * 0.2:
            evaluation["issues"].append(f"过多短句块: {too_short}个")
        
        if too_long > len(chunks) * 0.2:
            evaluation["issues"].append(f"过多长句块: {too_long}个")
        
        # 质量打分 (满分100)
        score = 100
        
        if too_short > len(chunks) * 0.1:
            score -= 20
        if too_long > len(chunks) * 0.1:
            score -= 20
        if evaluation["avg_chunk_length"] < 200:
            score -= 10
        
        evaluation["quality_score"] = max(score, 0)
        
        return evaluation
    
    @classmethod
    def generate_quality_report(cls, evaluation: Dict) -> str:
        """生成质量报告"""
        report = f"""
知识库质量评估报告
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
总文档块数：{evaluation['total_chunks']}
平均块长度：{evaluation['avg_chunk_length']:.0f}字符
质量评分：{evaluation['quality_score']}/100
"""
        
        if evaluation["issues"]:
            report += "\n⚠️ 发现的问题：\n"
            for issue in evaluation["issues"]:
                report += f"- {issue}\n"
        
        return report
```
